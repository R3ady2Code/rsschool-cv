function duplicateEncode(word) {
  const lettersArray = word.toLowerCase().split("");
  let result = "";
  lettersArray.map((letter, i) => {
    const arrWithoutL = (
      lettersArray
        .slice(0, i)
        .concat(lettersArray.slice(i + 1, lettersArray.length))
    );

    if (arrWithoutL.includes(letter)) {
      result += ")";
    } else {
      result += "(";
    }
  })

  return result
}
